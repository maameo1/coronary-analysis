\begin{frame}{MIUA --- The Question}

\begin{block}{Research Question}
How should we adapt a pretrained CT foundation model for topology-aware coronary segmentation?
\end{block}

\begin{itemize}
    \item \textbf{CT-FM}: SegResNet pretrained on 148K CT scans (Project LIGHTER)
    \item Fine-tune with \textbf{Soft BCS loss} (differentiable bifurcation connectivity)
    \item 5 adaptation strategies $\times$ 3 BCS weights = ablation grid
    \item Test on ImageCAS ($n=250$) + ASOCA cross-domain ($n=40$)
\end{itemize}

\vspace{0.3cm}
\textbf{Adaptation strategies}: Frozen, BitFit, Adapters, Partial unfreeze, Full unfreeze

\end{frame}

\vspace{0.3cm}
\begin{itemize}
    \item Threshold calibration alone recovers +18\pp{} Dice at no BCS cost
    \item $k=1$ (just 2 training scans!) is best --- more data overfits to sparse annotations
    \item Shallow decoder freeze preserves topology; full fine-tuning destroys it
\end{itemize}

\begin{frame}{TMI --- Architecture Findings}

\textbf{Top 4} (val Dice):
\begin{enumerate}
    \item \textbf{STU-Net Large} (0.801) --- brute-force capacity, 440M random-init params
    \item \textbf{MedNeXt-B} (0.800) --- modern efficient CNN, only 11M params
    \item \textbf{CT-FM SegResNet} (0.792) --- domain-specific SSL (143K cardiac CTs)
    \item \textbf{SwinUNETR} (0.787) --- only competitive transformer (hierarchical attention)
\end{enumerate}

\vspace{0.2cm}
\textbf{Key insights}:
\begin{itemize}
    \item \textbf{CNNs beat transformers} for fine vessels --- locality bias suits thin structures
    \item \textbf{Capacity or modern design} compensates for no pretraining
    \item \textbf{Wrong-domain pretraining hurts}: BTCV makes SwinUNETR ($-$1.3\pp) and UNETR ($-$1.6\pp) worse
    \item \textbf{Coarse supervised pretraining doesn't transfer}: VISTA3D (11.4K CTs) = 0.763
    \item \textbf{Hierarchical $\gg$ flat attention}: SwinUNETR (0.787) $\gg$ UNETR (0.760)
\end{itemize}

\vspace{0.2cm}
\warn{Caveat: these are val Dice only --- test inference running (Job 8139)}

\end{frame}
